导入说明 ​


安装 ​

javascript
npm install charoite-view --save-dev

全局注册 ​

javascript
import { createApp } from 'vue';
import CharoiteView from 'charoite-view';
import App from './App';
import 'ant-design-vue/dist/antd.css';

const app = createApp(App);

app.use(CharoiteView).mount('#app');

全局局部注册 ​

javascript
import { createApp } from 'vue';
import { CViewDetail } from 'charoite-view';
import App from './App';
import 'ant-design-vue/dist/antd.css';

const app = createApp(App);

app.use(CViewDetail).mount('#app');

组件内直接使用 ​

javascript
app.use(CViewDetail).mount('#app');

按需加载 ​

javascript
import CViewDetail from 'charoite-view/es/view-detail';
import 'charoite-view/es/view-detail/style.css'
